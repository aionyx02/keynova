### 1️⃣ CommandRouter：命令統一分發

**設計原理：**
所有功能都是「命令」，由中央路由器分發。官方功能和社群插件統一模式。

```rust
// src-tauri/src/core/command_router.rs

pub trait Command: Send + Sync {
    fn name(&self) -> &str;
    fn version(&self) -> &str;
    fn execute(&self, args: serde_json::Value) -> impl Future<Output = Result<serde_json::Value>>;
    fn description(&self) -> &str;
}

pub struct CommandRouter {
    commands: Arc<DashMap<String, Arc<dyn Command>>>,
    event_bus: Arc<EventBus>,
}

impl CommandRouter {
    pub fn new(event_bus: Arc<EventBus>) -> Self { ... }
    
    pub fn register(&self, cmd: Arc<dyn Command>) { ... }
    
    pub async fn execute(&self, name: &str, args: serde_json::Value) 
        -> Result<serde_json::Value> { ... }
    
    pub fn list_commands(&self) -> Vec<CommandInfo> { ... }
}

// 官方命令實現示例：
pub struct LaunchAppCommand {
    app_manager: Arc<AppManager>,
    event_bus: Arc<EventBus>,
}

impl Command for LaunchAppCommand {
    fn name(&self) -> &str { "launch_app" }
    fn execute(&self, args: serde_json::Value) -> impl Future<Output = Result<serde_json::Value>> {
        self.app_manager.launch(args["path"].as_str().unwrap())
    }
}
```

**v3.0 插件集成：**
```rust
// 加載社群插件（只需添加這段，無需改路由器）
pub fn load_plugin_commands(
    router: &CommandRouter,
    plugin_path: &Path
) -> Result<()> {
    let plugin_lib = libloading::Library::new(plugin_path)?;
    unsafe {
        let plugin_fn = plugin_lib.get::<fn() -> Arc<dyn Command>>(b"create_command")?;
        let cmd = plugin_fn();
        router.register(cmd);
    }
    Ok(())
}
```

### 2️⃣ EventBus：實時雙向推送

**設計原理：**
前端監聽事件，後端推送更新。支持終端I/O、搜索進度、通知等場景。

```rust
// src-tauri/src/core/event_bus.rs

pub enum Event {
    // 內置事件
    HotkeyTriggered(String),
    TerminalOutput(String, String),      // terminal_id, output
    SearchProgress(u32),                  // progress %
    ConfigReloaded,
    PluginLoaded(String),
    
    // 插件事件（可擴展）
    Custom(String, serde_json::Value),
}

pub struct EventBus {
    tx: tokio::sync::broadcast::Sender<Event>,
}

impl EventBus {
    pub fn new() -> Self { ... }
    
    pub fn emit(&self, event: Event) { let _ = self.tx.send(event); }
    
    pub fn subscribe(&self) -> tokio::sync::broadcast::Receiver<Event> {
        self.tx.subscribe()
    }
}

// 在 Tauri 中發送事件到前端
#[tauri::command]
async fn subscribe_to_events(window: tauri::Window) {
    let mut rx = APP_STATE.event_bus.subscribe();
    tokio::spawn(async move {
        while let Ok(event) = rx.recv().await {
            let _ = window.emit("app-event", event);
        }
    });
}
```

**前端監聽：**
```typescript
// src/hooks/useEvents.ts
export function useEvents(callback: (event: AppEvent) => void) {
    useEffect(() => {
        const unlisten = listen('app-event', (event) => {
            callback(event.payload);
        });
        return () => { unlisten.then(fn => fn()); };
    }, []);
}

// 使用示例
useEvents((event) => {
    if (event.type === 'TerminalOutput') {
        setTerminalOutput(prev => prev + event.output);
    } else if (event.type === 'SearchProgress') {
        setProgress(event.progress);
    }
});
```

### 3️⃣ ConfigManager：分層配置系統

**設計原理：**
配置分層存儲，支持主題、宏、插件定制。運行時熱加載，無需重啟應用。

```rust
// src-tauri/src/core/config_manager.rs

pub struct ConfigManager {
    config_dir: PathBuf,
    system_config: Arc<Mutex<SystemConfig>>,
    hotkey_config: Arc<Mutex<HotkeyConfig>>,
    theme_config: Arc<Mutex<ThemeConfig>>,
    macro_config: Arc<Mutex<MacroConfig>>,
}

impl ConfigManager {
    pub async fn load_all(&mut self) -> Result<()> {
        self.load_system_config().await?;
        self.load_hotkey_config().await?;
        self.load_theme_config().await?;
        self.load_macro_config().await?;
        Ok(())
    }
    
    pub async fn reload_theme(&mut self, theme_name: &str) -> Result<()> {
        // 動態加載主題，emit 事件通知前端重新渲染
        let theme = self.load_theme(theme_name).await?;
        *self.theme_config.lock().await = theme;
        self.event_bus.emit(Event::ConfigReloaded);
        Ok(())
    }
    
    pub async fn apply_macro(&self, macro_name: &str) -> Result<()> {
        // 執行宏序列
        let macro_def = self.macro_config.lock().await
            .macros.get(macro_name).unwrap();
        for step in &macro_def.steps {
            self.execute_macro_step(step).await?;
        }
        Ok(())
    }
}

// 配置數據結構
#[derive(Deserialize, Serialize)]
pub struct ThemeConfig {
    pub name: String,
    pub colors: HashMap<String, String>,
    pub fonts: HashMap<String, String>,
}

#[derive(Deserialize, Serialize)]
pub struct MacroConfig {
    pub macros: HashMap<String, MacroDef>,
}

pub struct MacroDef {
    pub name: String,
    pub trigger: String,  // Hotkey
    pub steps: Vec<MacroStep>,
}

pub struct MacroStep {
    pub command: String,
    pub args: serde_json::Value,
    pub wait_ms: Option<u32>,
}
```

### 4️⃣ SearchRegistry：統一索引層

**設計原理：**
所有可搜索的數據源（文件、應用、歷史、剪貼簿）都通過統一索引器註冊。v2.0+ 擴展時只需添加新的 indexer。

```rust
// src-tauri/src/core/search_registry.rs

pub trait Indexer: Send + Sync {
    fn name(&self) -> &str;
    fn index(&self) -> Result<Vec<SearchableItem>>;
    fn search(&self, query: &str, limit: usize) -> Result<Vec<SearchResult>>;
    fn refresh(&self) -> Result<()>;  // 增量更新
}

pub struct SearchRegistry {
    indexers: Arc<DashMap<String, Arc<dyn Indexer>>>,
    event_bus: Arc<EventBus>,
}

impl SearchRegistry {
    pub fn register_indexer(&self, indexer: Arc<dyn Indexer>) {
        self.indexers.insert(indexer.name().to_string(), indexer);
    }
    
    pub async fn unified_search(
        &self,
        query: &str,
        target_indexers: Vec<&str>,
        limit: usize,
    ) -> Result<Vec<SearchResult>> {
        let mut results = Vec::new();
        let mut tasks = vec![];
        
        // 並行搜索多個索引源
        for indexer_name in target_indexers {
            if let Some(indexer) = self.indexers.get(indexer_name) {
                let indexer = indexer.clone();
                let query = query.to_string();
                let event_bus = self.event_bus.clone();
                
                tasks.push(tokio::spawn(async move {
                    event_bus.emit(Event::SearchProgress(0));
                    let mut res = indexer.search(&query, limit).unwrap_or_default();
                    event_bus.emit(Event::SearchProgress(100));
                    res
                }));
            }
        }
        
        // 等待所有搜索完成
        for task in tasks {
            let res = task.await?;
            results.extend(res);
        }
        
        // 按相關度排序
        results.sort_by(|a, b| b.relevance.partial_cmp(&a.relevance).unwrap());
        Ok(results.into_iter().take(limit).collect())
    }
}

// 官方索引器示例
pub struct FileIndexer {
    tantivy_index: TantivyIndex,
    event_bus: Arc<EventBus>,
}

impl Indexer for FileIndexer {
    fn name(&self) -> &str { "files" }
    
    fn search(&self, query: &str, limit: usize) -> Result<Vec<SearchResult>> {
        let results = self.tantivy_index.search(query, limit)?;
        Ok(results.into_iter().map(|r| SearchResult {
            r#type: "file".to_string(),
            title: r.filename,
            path: r.path,
            relevance: r.score,
        }).collect())
    }
    
    fn refresh(&self) -> Result<()> {
        // 增量索引更新
        self.event_bus.emit(Event::SearchProgress(50));
        self.tantivy_index.refresh()?;
        self.event_bus.emit(Event::SearchProgress(100));
        Ok(())
    }
}
```
